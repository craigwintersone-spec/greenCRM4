// ── AUTH (queries super_admins table) ── async function isSuperAdmin(userId){ if(!userId) return false; const { data, error } = await sb.from('super_admins').select('user_id').eq('user_id', userId).maybeSingle(); if(error){ console.warn('super_admins check failed:', error.message); return false; } return !!data; } async function boot(){ const { data: { session } } = await sb.auth.getSession(); if(session){ const ok = await isSuperAdmin(session.user.id); if(!ok){ showAuthErr('This account does not have super admin access.'); await sb.auth.signOut(); // log them out so they can't loiter on the auth screen logged in return; } currentUser = session.user; showApp(); } // else stay on auth screen } function showApp(){ $('auth-screen').style.display='none'; $('app-screen').classList.add('visible'); $('user-email').textContent=currentUser.email; loadOrgs(); } function showAuthErr(msg){ const e = $('si-err'); e.textContent = msg; e.style.display = 'block'; } async function doSignIn(){ const email = $('si-email').value.trim(), pass = $('si-pass').value; $('si-err').style.display = 'none'; if(!email || !pass){ showAuthErr('Please enter your email and password.'); return; } const btn = $('si-btn'); btn.textContent = 'Signing in…'; btn.disabled = true; const { data, error } = await sb.auth.signInWithPassword({ email, password: pass }); btn.textContent = 'Sign in to Admin'; btn.disabled = false; if(error){ showAuthErr(error.message); return; } // Now check super_admins membership — single source of truth const ok = await isSuperAdmin(data.user.id); if(!ok){ showAuthErr('This account does not have super admin access.'); await sb.auth.signOut(); return; } currentUser = data.user; showApp(); } async function doSignOut(){ await sb.auth.signOut(); window.location.href = 'index.html'; } document.addEventListener('keydown', e => { if(e.key === 'Enter' && $('auth-screen').style.display !== 'none') doSignIn(); });
